		public unsafe void Main()
		{
			CoverageProbe coverageProbe = CoverageReporter<ModuleCoverage>.AcquireFileCounter(0);
			try
			{
				int* pointer = (int*)coverageProbe.Pointer;
				pointer[7]++;
				pointer[8]++;
				Console.WriteLine("Main Method");
				pointer[9]++;
				int num = 0;
				while (true)
				{
					pointer[9]++;
					if (num >= 100)
					{
						break;
					}
					pointer[10]++;
					pointer[11]++;
					Console.WriteLine(num);
					pointer[12]++;
					num++;
				}
				pointer[14]++;
				MyProperty = "Value";
				pointer[16]++;
				Console.WriteLine(MyProperty);
				pointer[17]++;
			}
			catch
			{
				//try-fault
				coverageProbe.Dispose();
				throw;
			}
			coverageProbe.Dispose();
		}

		public async Task MainAsync()
		{
			await Task.Delay(100);
		}
	}
